# All-Lines Tracking Presentation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Give every supported line the Seoul Line 2 presentation — one persistent SVG world with a tracking camera and train, the balanced green platform sign, and the character-feedback typing field — then delete the focused eight-station renderer it replaces.

**Architecture:** A new `lineTopology.ts` resolves a run to `{key, sequence, path, loop}` using the existing source-guided polylines in `routeGeometry.ts`. A new `lineWorld.ts` turns that into a persistent world: the control polyline is corner-rounded and scaled **once, over the whole resolved sequence**, so arc length per station is constant even across a junction such as 구로; the rendered path and camera are then clipped to the run's own stations. `Line2TrackingMap` / `Line2TypingField` are generalized and renamed rather than forked.

**Tech Stack:** Vite, React 19, TypeScript (strict), Vitest, custom CSS/SVG. No new runtime dependencies.

## Global Constraints

- Spec: `docs/superpowers/specs/2026-07-23-all-lines-tracking-design.md`.
- Station **order** stays on the frozen official data in `src/data/lines.ts`. New data supplies English name and station number only.
- Station info is keyed `lineId → korean → { korean, english, number }`; one station has a different number per line (서울역 = `133` / `426` / `A01`).
- Arc length per station is one constant across a whole resolved sequence. Never rescale per leg or per segment.
- Input, scoring, and submission never wait on camera or train state.
- Reduced motion settles camera and train immediately and removes halo translation.
- `any` and `unknown` are forbidden. Run `npx tsc -p tsconfig.json --noEmit` after every task.
- Match the existing dense house style in `src/game` and `src/components`: no semicolons, minimal spacing, arrow consts.
- Quality gate is `npm run check` (ESLint, Vitest, strict TypeScript, production build).
- Do not delete `RouteMap`/`labels.ts`/`randomRoute.ts` before Task 9.

Create `src/game/lineTopology.ts`:

```ts
import type { Point } from './geometry'
import { getLine } from '../data/lines'
import { baseRoutes,line1Legs,topologyRoutes } from './routeGeometry'

export type Topology={key:string;sequence:string[];path:Point[];loop:boolean}
type Candidate={key:string;sequence:string[];path:Point[]}

const join=(head:string[],tail:string[])=>[...head,...tail.slice(1)]
const joinPath=(head:Point[],tail:Point[])=>[...head,...tail.slice(1)]

function candidatesFor(lineId:string):Candidate[]{
  const line=getLine(lineId)
  if(lineId==='seoul-1'){
    const [north,incheon,sinchang]=line.sequences as [string[],string[],string[]]
    const [northLeg,incheonLeg,sinchangLeg]=line1Legs as [Point[],Point[],Point[]]
    const reverse=<T,>(items:T[])=>[...items].reverse()
    return [
      // Cross-Guro joins first: a run spanning two legs must not match a single leg.
      {key:'seoul-1-incheon-north',sequence:join(reverse(incheon),reverse(north)),path:joinPath(reverse(incheonLeg),reverse(northLeg))},
      {key:'seoul-1-north-incheon',sequence:join(north,incheon),path:joinPath(northLeg,incheonLeg)},
      {key:'seoul-1-sinchang-north',sequence:join(reverse(sinchang),reverse(north)),path:joinPath(reverse(sinchangLeg),reverse(northLeg))},
      {key:'seoul-1-north-sinchang',sequence:join(north,sinchang),path:joinPath(northLeg,sinchangLeg)},
      {key:'seoul-1-incheon-sinchang',sequence:join(reverse(incheon),sinchang),path:joinPath(reverse(incheonLeg),sinchangLeg)},
      {key:'seoul-1-sinchang-incheon',sequence:join(reverse(sinchang),incheon),path:joinPath(reverse(sinchangLeg),incheonLeg)},
      {key:'seoul-1-north',sequence:north,path:northLeg},
      {key:'seoul-1-incheon',sequence:incheon,path:incheonLeg},
      {key:'seoul-1-sinchang',sequence:sinchang,path:sinchangLeg},
    ]
  }
  if(lineId==='seoul-5'){
    const [trunk,hanam,macheon]=line.sequences as [string[],string[],string[]]
    const reverse=<T,>(items:T[])=>[...items].reverse()
    return [
      {key:'seoul-5-hanam-macheon',sequence:join(reverse(hanam),macheon),path:topologyRoutes['seoul-5-hanam-macheon']!.path},
      {key:'seoul-5-macheon-hanam',sequence:join(reverse(macheon),hanam),path:topologyRoutes['seoul-5-macheon-hanam']!.path},
      {key:'seoul-5-hanam',sequence:join(trunk,hanam),path:topologyRoutes['seoul-5-hanam']!.path},
      {key:'seoul-5-macheon',sequence:join(trunk,macheon),path:topologyRoutes['seoul-5-macheon']!.path},
      {key:'seoul-5-trunk',sequence:trunk,path:topologyRoutes['seoul-5-trunk']!.path},
    ]
  }
  if(lineId==='seoul-6')return [
    {key:'seoul-6-loop',sequence:line.oneWaySequences![0]!.slice(0,-1),path:topologyRoutes['seoul-6-loop']!.path},
    {key:'seoul-6-trunk',sequence:line.sequences[0]!,path:topologyRoutes['seoul-6-trunk']!.path},
  ]
  if(lineId==='seoul-9')return (line.services??[]).map(service=>({
    key:`seoul-9-${service.id}`,sequence:[...service.sequence],path:topologyRoutes['seoul-9']!.path,
  }))
  const path=topologyRoutes[lineId]?.path??baseRoutes[lineId]
  if(!path)throw new Error(`Missing route geometry: ${lineId}`)
  return line.sequences.map((sequence,index)=>({key:index?`${lineId}-${index}`:lineId,sequence,path}))
}

const startsAt=(sequence:string[],stations:string[])=>{
  const start=sequence.indexOf(stations[0]!)
  return start>=0&&stations.every((station,offset)=>sequence[start+offset]===station)?start:-1
}
const rotate=(sequence:string[],start:number)=>sequence.map((_,offset)=>sequence[(start+offset)%sequence.length]!)

export function resolveTopology(lineId:string,stations:string[]):Topology{
  const line=getLine(lineId)
  const candidates=candidatesFor(lineId)
  if(line.loop){
    const {key,sequence,path}=candidates[0]!
    const forwardStart=sequence.indexOf(stations[0]!)
    if(forwardStart<0)throw new Error(`Unknown station on ${lineId}: ${stations[0]}`)
    const reversed=stations.length>1&&sequence[(forwardStart+1)%sequence.length]!==stations[1]
    const oriented=reversed?rotate([...sequence].reverse(),sequence.length-1-forwardStart):rotate(sequence,forwardStart)
    return {key,sequence:oriented,path:reversed?[...path].reverse():path,loop:true}
  }
  for(const candidate of candidates){
    if(startsAt(candidate.sequence,stations)>=0)return {...candidate,loop:false}
    const sequence=[...candidate.sequence].reverse()
    if(startsAt(sequence,stations)>=0)return {key:candidate.key,sequence,path:[...candidate.path].reverse(),loop:false}
  }
  throw new Error(`Unresolvable run on ${lineId}: ${stations[0]} → ${stations.at(-1)}`)
}
```

`topologyRoutes['seoul-9']` already exists in `routeGeometry.ts` and covers both service patterns, so local and express share one control polyline and differ only in their sequence.

Create `src/game/lineWorld.ts`:

```ts
import type { Point } from './geometry'
import { resolveTopology } from './lineTopology'

export const STATION_SPACING=77
const CORNER=0.28
const SAMPLES_PER_SEGMENT=24

export type LineWorld={
  key:string
  loop:boolean
  pathD:string
  stationNames:string[]
  stationDistances:number[]
  pointAt(distance:number):{x:number;y:number;angle:number}
  cameraWidth(index:number):number
}
type Sample={x:number;y:number;distance:number}

const lerp=(a:Point,b:Point,t:number):Point=>[a[0]+(b[0]-a[0])*t,a[1]+(b[1]-a[1])*t]
const quadratic=(a:Point,b:Point,c:Point,t:number):Point=>{
  const u=1-t
  return [u*u*a[0]+2*u*t*b[0]+t*t*c[0],u*u*a[1]+2*u*t*b[1]+t*t*c[1]]
}

// Round every interior corner so the world reads as track rather than as a chevron.
function roundCorners(points:Point[],closed:boolean):Point[]{
  if(points.length<3)return points
  const result:Point[]=[]
  const count=closed?points.length:points.length-1
  if(!closed)result.push(points[0]!)
  for(let index=closed?0:1;index<count;index++){
    const previous=points[(index-1+points.length)%points.length]!
    const current=points[index]!
    const next=points[(index+1)%points.length]!
    const entry=lerp(current,previous,CORNER)
    const exit=lerp(current,next,CORNER)
    result.push(entry)
    for(let step=1;step<=SAMPLES_PER_SEGMENT;step++)result.push(quadratic(entry,current,exit,step/SAMPLES_PER_SEGMENT))
  }
  if(!closed)result.push(points.at(-1)!)
  else result.push(result[0]!)
  return result
}

function sample(points:Point[]):Sample[]{
  const samples:Sample[]=[{x:points[0]![0],y:points[0]![1],distance:0}]
  for(let index=1;index<points.length;index++){
    const previous=samples.at(-1)!
    const [x,y]=points[index]!
    samples.push({x,y,distance:previous.distance+Math.hypot(x-previous.x,y-previous.y)})
  }
  return samples
}

const scaleSamples=(samples:Sample[],factor:number):Sample[]=>samples.map(item=>({x:item.x*factor,y:item.y*factor,distance:item.distance*factor}))

export function getLineWorld(lineId:string,stations:string[]):LineWorld{
  const topology=resolveTopology(lineId,stations)
  const rounded=roundCorners(topology.path.map(([x,y])=>[x,y] as Point),topology.loop)
  const raw=sample(rounded)
  const worldStations=topology.loop?topology.sequence.length:topology.sequence.length-1
  const wanted=STATION_SPACING*Math.max(1,worldStations)
  const samples=scaleSamples(raw,wanted/(raw.at(-1)!.distance||1))
  const total=samples.at(-1)!.distance

  const pointAt=(distance:number)=>{
    const target=topology.loop?((distance%total)+total)%total:Math.min(total,Math.max(0,distance))
    let low=0,high=samples.length-1
    while(low<high){const middle=Math.floor((low+high)/2);if(samples[middle]!.distance<target)low=middle+1;else high=middle}
    const next=samples[low]!,previous=samples[Math.max(0,low-1)]!
    const span=next.distance-previous.distance||1,ratio=(target-previous.distance)/span
    return {
      x:previous.x+(next.x-previous.x)*ratio,
      y:previous.y+(next.y-previous.y)*ratio,
      angle:Math.atan2(next.y-previous.y,next.x-previous.x)*180/Math.PI,
    }
  }

  const offset=topology.sequence.indexOf(stations[0]!)
  const stationDistances=stations.map((_,index)=>(offset+index)*STATION_SPACING)

  const cameraWidth=(index:number)=>{
    const nearby=stationDistances.slice(Math.max(0,index-2),index+3).map(pointAt)
    const xs=nearby.map(point=>point.x),ys=nearby.map(point=>point.y)
    const span=Math.max(Math.max(...xs)-Math.min(...xs),(Math.max(...ys)-Math.min(...ys))*1.55)
    return Math.max(360,Math.min(620,span*1.45+150))
  }

  const from=stationDistances[0]!,to=stationDistances.at(-1)!
  const step=(to-from)/Math.max(1,(stations.length-1)*SAMPLES_PER_SEGMENT)
  const drawn=stations.length>1
    ?Array.from({length:(stations.length-1)*SAMPLES_PER_SEGMENT+1},(_,index)=>pointAt(from+index*step))
    :[pointAt(from),pointAt(from)]
  const pathD=drawn.map((point,index)=>`${index?'L':'M'} ${point.x.toFixed(3)} ${point.y.toFixed(3)}`).join(' ')

  return {key:topology.key,loop:topology.loop,pathD,stationNames:[...stations],stationDistances,pointAt,cameraWidth}
}
```
